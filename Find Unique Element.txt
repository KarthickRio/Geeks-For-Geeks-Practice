class Solution {
    
    public int findUnique(int a[], int n, int k)
    {
       HashMap <Integer,Integer> map = new HashMap<>();
       for(int i = 0; i<n;i++)
       {
           map.put(a[i],map.getOrDefault(a[i],0)+1);
           
       }
       for(int i = 0; i<n;i++)
       {
           if(map.get(a[i])%k !=0)
           {
               return a[i];
           }
           
       }
       return -1;
    }
}
